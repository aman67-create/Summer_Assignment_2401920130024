class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int profit = 0;
        int curr = prices[0];
        int n = prices.size();

        for(int i = 1; i<n; i++){
            if(prices[i]<curr)
            curr = prices[i];

            profit = max((prices[i]-curr),profit);
        }

        return profit;
    }
};
